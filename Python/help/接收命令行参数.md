在Python中，接收参数主要有两种情况：接收命令行参数和接收函数参数。下面是详细的解释和示例：

## 1. 接收命令行参数

### 1.1 使用 `sys.argv`（简单场景）

```python
import sys

# 示例：python script.py arg1 arg2 arg3
if __name__ == "__main__":
    # sys.argv[0] 是脚本名称
    # sys.argv[1:] 是传递的参数
    print("脚本名称:", sys.argv[0])
    print("参数列表:", sys.argv[1:])
    print("参数数量:", len(sys.argv) - 1)
    
    # 处理参数
    for i, arg in enumerate(sys.argv[1:], 1):
        print(f"参数 {i}: {arg}")
```

### 1.2 使用 `argparse`模块（推荐）

```python
import argparse

# 创建解析器
parser = argparse.ArgumentParser(description='示例脚本')

# 添加参数
parser.add_argument('input', help='输入文件路径')
parser.add_argument('-o', '--output', help='输出文件路径', default='output.txt')
parser.add_argument('-n', '--number', type=int, help='数字参数', default=10)
parser.add_argument('-v', '--verbose', action='store_true', help='详细模式')
parser.add_argument('--mode', choices=['fast', 'slow'], default='fast', help='运行模式')

# 解析参数
args = parser.parse_args()

# 使用参数
print(f"输入文件: {args.input}")
print(f"输出文件: {args.output}")
print(f"数字: {args.number}")
print(f"详细模式: {args.verbose}")
print(f"模式: {args.mode}")
```

## 2. 接收函数参数

### 2.1 基本参数

```python
# 位置参数
def greet(name, greeting):
    return f"{greeting}, {name}!"

print(greet("Alice", "Hello"))  # Hello, Alice!

# 默认参数
def power(base, exponent=2):
    return base ** exponent

print(power(3))      # 9
print(power(3, 3))   # 27
```

### 2.2 可变参数

```python
# *args - 接收任意数量的位置参数（元组）
def sum_all(*args):
    return sum(args)

print(sum_all(1, 2, 3, 4))  # 10

# **kwargs - 接收任意数量的关键字参数（字典）
def print_info(**kwargs):
    for key, value in kwargs.items():
        print(f"{key}: {value}")

print_info(name="Alice", age=25, city="New York")

# 混合使用
def func(a, b, *args, **kwargs):
    print(f"a={a}, b={b}")
    print(f"args={args}")
    print(f"kwargs={kwargs}")

func(1, 2, 3, 4, 5, x=10, y=20)
```

### 2.3 参数解包

```python
def process_data(name, value, mode="normal"):
    print(f"{name}: {value} ({mode})")

# 列表/元组解包
args = ["data1", 100]
process_data(*args)  # 相当于 process_data("data1", 100)

# 字典解包
kwargs = {"name": "data2", "value": 200, "mode": "fast"}
process_data(**kwargs)  # 相当于 process_data(name="data2", value=200, mode="fast")
```

## 3. 完整示例

```python
#!/usr/bin/env python3
"""
文件处理脚本示例
用法: python script.py input.txt -o output.txt -v
"""

import argparse
import sys

def process_file(input_file, output_file, verbose=False):
    """处理文件的函数"""
    if verbose:
        print(f"开始处理文件: {input_file}")
    
    try:
        with open(input_file, 'r') as f:
            content = f.read()
        
        # 处理内容（示例：转换为大写）
        processed = content.upper()
        
        with open(output_file, 'w') as f:
            f.write(processed)
            
        if verbose:
            print(f"处理完成，保存到: {output_file}")
        return True
    except Exception as e:
        print(f"错误: {e}")
        return False

def main():
    parser = argparse.ArgumentParser(description='文件处理工具')
    parser.add_argument('input', help='输入文件路径')
    parser.add_argument('-o', '--output', help='输出文件路径')
    parser.add_argument('-v', '--verbose', action='store_true', 
                       help='显示详细输出')
    parser.add_argument('--encoding', default='utf-8',
                       help='文件编码 (默认: utf-8)')
    
    args = parser.parse_args()
    
    # 如果未指定输出文件，则自动生成
    if not args.output:
        import os
        name, ext = os.path.splitext(args.input)
        args.output = f"{name}_processed{ext}"
    
    # 调用处理函数
    success = process_file(
        input_file=args.input,
        output_file=args.output,
        verbose=args.verbose
    )
    
    return 0 if success else 1

if __name__ == "__main__":
    sys.exit(main())
```

## 4. 参数验证

```python
def validate_parameters(func):
    """参数验证装饰器"""
    def wrapper(*args, **kwargs):
        # 参数验证逻辑
        for arg in args:
            if arg is None:
                raise ValueError("参数不能为None")
        return func(*args, **kwargs)
    return wrapper

@validate_parameters
def safe_function(name, value):
    print(f"{name}: {value}")

# 使用类型提示
from typing import Optional

def typed_function(name: str, age: int, city: Optional[str] = None) -> str:
    """带类型提示的函数"""
    if city:
        return f"{name}, {age}岁, 来自{city}"
    return f"{name}, {age}岁"
```

## 总结

1. **命令行参数**： 简单场景用 `sys.argv` 复杂场景用 `argparse` Web/API场景考虑 `click`
2. **函数参数**： 位置参数、关键字参数 默认参数值 可变参数（`*args`, `**kwargs`）
3. **最佳实践**： 为命令行工具提供帮助文档 设置合理的默认值 进行参数验证 使用类型提示提高可读性
